\begin{column}{\colwidth}

  \begin{alertblock}{Problema e Oportunidade}
    Crianças com baixa visão enfrentam barreiras em brincadeiras educativas que
    dependem fortemente de estimulos visuais. A proposta da Fazzle e ampliar
    acessibilidade e autonomia por meio de um quebra-cabeça multissensorial.
  \end{alertblock}

  \begin{block}{Proposta de Valor}
    \textbf{Como contribuimos?}

    \begin{itemize}
      \item Promovemos desenvolvimento neuro-motor, memória e coordenação.
      \item Combinamos alto relevo, pecas magnéticas e feedback sonoro.
      \item Transformamos o aprendizado em experiência inclusiva e lúdica.
    \end{itemize}
  \end{block}

  \begin{block}{Segmento de Clientes}
    \textbf{Quem ajudamos?}

    \begin{itemize}
      \item Crianças de 5 a 10 anos com algum nível de deficiência visual.
      \item Pais e responsaveis que buscam recursos educativos inclusivos.
      \item Escolas e terapeutas como influenciadores de adoção.
    \end{itemize}
  \end{block}

  \begin{block}{Descrição da Solução}
    Produto educacional em formato de quebra-cabeça com arquitetura voltada para
    habilidades cognitivas e motoras. O sistema RFID detecta tentativas de
    encaixe e retorna dicas sonoras quando necessário.
  \end{block}

\end{column}

\separatorcolumn

\begin{column}{\colwidth}

  \begin{block}{Atividades Chave}
    \textbf{O que fazemos?}

    \begin{itemize}
      \item Desenvolvimento de jogo educativo adaptado para baixa visão.
      \item Integração de hardware (RFID e base magnética) ao tabuleiro.
      \item Curadoria pedagogica focada em memória e coordenação.
      \item Testes de usabilidade com publico infantil e familiares.
    \end{itemize}
  \end{block}

  \begin{block}{Recursos Principais}
    \textbf{Quem somos e o que temos?}

    \begin{itemize}
      \item Equipe especializada em desenvolvimento de jogos.
      \item Ênfase em temática ambiental e impacto social.
      \item Recursos computacionais e prototipagem física.
      \item App nativo Android/iOS para suporte e acompanhamento.
    \end{itemize}
  \end{block}

  \begin{block}{Parcerias Principais}
    \textbf{Quem ajuda?}

    \begin{itemize}
      \item Produtores de insumos de informática e componentes eletrônicos.
      \item Fornecedores de embalagens e bioplasticos.
      \item Influenciadores e parceiros de marketing digital.
      \item Empresas da cadeia quimica para materiais sustentáveis.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Canais e Relacionamento}
    \textbf{Como nos encontram?}

    \begin{itemize}
      \item E-commerce proprio.
      \item Redes sociais e trafego pago.
      \item Pontos de venda com estratégia B2C.
    \end{itemize}

    \textbf{Como interagimos?}

    \begin{itemize}
      \item Atendimento digital contínuo no pre e pos-venda.
      \item Conteúdos educativos para pais e crianças.
      \item Suporte orientado ao uso do sistema RFID.
    \end{itemize}
  \end{exampleblock}

\end{column}

  \begin{block}{Estrutura de Custos}
    \textbf{Principais elementos de custo}

    \begin{itemize}
      \item Desenvolvimento e manutenção de app Android/iOS.
      \item Sistema RFID e tabuleiro magnético.
      \item Materiais de embalagem em bioplastico.
      \item Operação de e-commerce e aquisição de clientes.
    \end{itemize}
  \end{block}

  \begin{block}{Fontes de Receita e Beneficios}
    \textbf{Como monetizamos}

    \begin{itemize}
      \item Venda direta do produto em e-commerce e redes sociais.
      \item Escalabilidade por canais digitais com alcance nacional.
      \item Benefício social: inclusão e desenvolvimento infantil.
    \end{itemize}
  \end{block}

  \begin{block}{Resumo do Modelo de Negócio}
    A Fazzle conecta \textbf{tecnologia assistiva}, \textbf{educacao infantil}
    e \textbf{canais digitais} para entregar um produto acessível com impacto
    social e potencial de sustentabilidade financeira.
  \end{block}
